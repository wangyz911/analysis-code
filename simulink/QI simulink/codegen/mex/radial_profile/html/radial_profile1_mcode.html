1function [cenX,cenY] = radial_profile(img_array,xc,yc,r_step,theta_num_perQ)
 2% 本函数用于计算若干张图像数据的质心位置，图像有先后顺序但无逻辑顺序。
 3% 本函数对图心的寻找进行了改进，主要是消除背景噪声。
 4% 输入来的是N张图拼接起来的长图矩阵。
 5% 本函数是labview 并行计划的第一环
 6% 本函数必须假设图像矩阵是长宽一致的。
 7
 8% 第一步，确定图像总矩阵大小，img_array 是一个三维矩阵，
 9img_array = im2double(img_array);
10 ROI_size = size(img_array,2);
11 % 第二步，按参数生成QI的采样极坐标，并生成极坐标矩阵
12 % 此处令r_max 等于距离边界最小的
13 r_max = floor(min([xc,yc,(ROI_size-xc),(ROI_size-yc)]))-1;
14%  r_max = 10;
15 r = 0:r_step:r_max;
16 theta_step = pi/2/theta_num_perQ;
17 theta = theta_step:theta_step:2*pi;
18 
19 r_matrix = zeros(size(theta,2),size(r,2));
20 theta_matrix = zeros(size(r,2),size(theta,2));
21 
22 for i = 1:size(theta,2)
23     r_matrix(i,:) = r;
24 end
25 
26  for i = 1:size(r,2)
27     theta_matrix(i,:) = theta;
28 end
29 theta_matrix = theta_matrix';
30 
31% 第三步，
32 radial_result = zeros(size(theta,2),size(r,2));
33for i = 1:size(theta,2)
34    for j = 1:size(r,2)
35        x = xc+r_matrix(i,j)*cos(theta_matrix(i,j));
36        y = yc+r_matrix(i,j)*sin(theta_matrix(i,j));
37        x1=floor(x);
38        y1=floor(y);
39        x2=x1+1;
40        y2=y1+1;
41        radial_result(i,j) = img_array(x1,y1)*(x2-x)*(y2-y)+img_array(x1,y2)*(x2-x)*(y-y1)...
42            +img_array(x2,y1)*(x-x1)*(y2-y)+img_array(x2,y2)*(x-x1)*(y-y1);
43    end
44end
45radial_sum = radial_result;
46% 将计算结果按四个象限平均。
47 radialProfile1 = mean(radial_sum(1:theta_num_perQ,:));
48 radialProfile2 = mean(radial_sum((theta_num_perQ+1):2*theta_num_perQ,:));
49 radialProfile3 = mean(radial_sum((2*theta_num_perQ+1):3*theta_num_perQ,:));
50 radialProfile4 = mean(radial_sum((3*theta_num_perQ+1):4*theta_num_perQ,:));
51 % 分别计算X Y 方向的相关曲线，及其下标
52 radial_left = radialProfile2+radialProfile3;
53 radial_right = radialProfile1+radialProfile4;
54 radial_x = [flip(radial_left(2:end),2),radial_right];
55 radial_fx = flip(radial_x,2);
56 corr_X = xcorr(radial_x,radial_fx);
57 
58 radial_up = radialProfile1+radialProfile2;
59 radial_down = radialProfile3+radialProfile4;
60 radial_y = [flip(radial_up(2:end),2),radial_down];
61 radial_fy = flip(radial_y,2);
62 corr_Y = xcorr(radial_y,radial_fy);
63 % 下标换算
64 size_radial = size(radial_x,2);
65 r_index = -(size_radial-1):(size_radial-1);
66 % 寻找相关曲线最高点，拟合计算偏移量
67 peak_x = find(corr_X==max(corr_X));
68 x_fitX = r_index((peak_x-3):(peak_x+3));
69 y_fitX =corr_X((peak_x-3):(peak_x+3));
70 px = polyfit(x_fitX,y_fitX,2);
71 x_shift = -px(2)/(2*px(1))*2/pi;
72 %
73 cenX = xc+x_shift;
74 
75 peak_y = find(corr_Y==max(corr_Y));
76 x_fitY = r_index((peak_y-3):(peak_y+3));
77 y_fitY =corr_Y((peak_y-3):(peak_y+3));
78 py = polyfit(x_fitY,y_fitY,2);
79 y_shift = -py(2)/(2*py(1))*2/pi;
80 % 下面是减号，因为matlab 和labview 的坐标原点不一样
81 cenY = yc-y_shift;
82     
83end
